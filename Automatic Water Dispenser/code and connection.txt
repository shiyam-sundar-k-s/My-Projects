#include <Servo.h>

#define DISP_TRIG 2
#define DISP_ECHO 3
#define RELAY_PIN 4

#define BIN_TRIG 6
#define BIN_ECHO 7
#define SERVO_PIN 9

Servo binServo;

void setup() {
  pinMode(DISP_TRIG, OUTPUT);
  pinMode(DISP_ECHO, INPUT);
  pinMode(RELAY_PIN, OUTPUT);

  pinMode(BIN_TRIG, OUTPUT);
  pinMode(BIN_ECHO, INPUT);

  binServo.attach(SERVO_PIN);
  binServo.write(0); // Dustbin closed
  digitalWrite(RELAY_PIN, LOW); // Pump OFF

  Serial.begin(9600);
}

void loop() {
  // --- Water Dispenser ---
  long dist1 = getDistance(DISP_TRIG, DISP_ECHO);
  if (dist1 < 10) {
    digitalWrite(RELAY_PIN, HIGH); // Turn ON Pump
    Serial.println("Water Dispensing...");
    delay(3000); // Dispense time
    digitalWrite(RELAY_PIN, LOW); // Turn OFF Pump
  }

  // --- Auto Dustbin ---
  long dist2 = getDistance(BIN_TRIG, BIN_ECHO);
  if (dist2 < 15) {
    binServo.write(90); // Open bin
    Serial.println("Bin Opened");
    delay(3000);
    binServo.write(0); // Close bin
    Serial.println("Bin Closed");
  }

  delay(500);
}

// Function to get distance from ultrasonic sensor
long getDistance(int trigPin, int echoPin) {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH);
  long distance = duration * 0.034 / 2;
  return distance;
}
